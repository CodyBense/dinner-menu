CREATE TABLE menu (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT,
        cuisine_type TEXT,
        flavor TEXT,
        difficulty TEXT,
        time INTEGER,
        liked INTEGER,
        link TEXT,
        made INTEGER,
        FOREIGN KEY(id, name, cuisine_type, flavor, difficulty, time, liked, link)
            REFERENCES recipes (id, name, cuisine_type, flavor, difficulty, time, liked, link)
)
